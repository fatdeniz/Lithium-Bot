<script>
    lucide.createIcons();

    // App State
    let chats = ["Technical Discussion", "Creative Writing", "Research Draft"];
    let username = "User";

    function switchView(viewId, el) {
        // Toggle visibility
        document.querySelectorAll('.view-section').forEach(v => v.classList.remove('active'));
        document.getElementById(viewId + '-view').classList.add('active');
        
        // Toggle nav items
        document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('active'));
        el.classList.add('active');
    }

    function renderHistory() {
        const list = document.getElementById('history-list');
        list.innerHTML = '';
        chats.forEach((title, i) => {
            const item = document.createElement('div');
            item.className = 'chat-entry';
            item.innerHTML = `<span><i data-lucide="message-circle" style="width:14px; vertical-align:middle; margin-right:8px;"></i>${title}</span>`;
            
            // Swipe to delete
            let startX;
            item.addEventListener('touchstart', e => startX = e.touches[0].clientX);
            item.addEventListener('touchend', e => {
                if (startX - e.changedTouches[0].clientX > 80) {
                    item.classList.add('swiping');
                    setTimeout(() => { chats.splice(i, 1); renderHistory(); }, 300);
                }
            });
            list.appendChild(item);
        });
        lucide.createIcons();
    }

    async function handleSend() {
        const input = document.getElementById('user-input');
        const flow = document.getElementById('message-flow');
        const mode = document.getElementById('mode-select').value;
        const text = input.value.trim();

        if (!text) return;
        
        // User Message
        appendBubble(text, 'user-bubble');
        input.value = '';

        // Response Logic
        setTimeout(() => {
            if (mode === 'image') {
                const imgUrl = `https://pollinations.ai/p/${encodeURIComponent(text)}?width=1024&height=1024&nologo=true`;
                appendBubble(`Lithium: Generated your request.<br><img src="${imgUrl}" class="result-img">`, 'lithium-bubble');
            } else {
                let response = "Lithium: Analysis complete. How would you like to proceed?";
                if (mode === 'voltage') response = "Lithium: Code refactored and optimized. Terminal ready.";
                if (mode === 'research') response = "Lithium: I've cross-referenced current data nodes. Here are the findings.";
                appendBubble(response, 'lithium-bubble');
            }
        }, 700);
    }

    function appendBubble(content, className) {
        const bubble = document.createElement('div');
        bubble.className = `msg-bubble ${className}`;
        bubble.innerHTML = content;
        const flow = document.getElementById('message-flow');
        flow.appendChild(bubble);
        flow.scrollTop = flow.scrollHeight;
    }

    function toggleSettings() {
        const overlay = document.getElementById('modal-overlay');
        overlay.style.display = overlay.style.display === 'flex' ? 'none' : 'flex';
    }

    function applySettings() {
        username = document.getElementById('set-name').value || "User";
        const theme = document.getElementById('set-theme').value;
        const persona = document.getElementById('set-persona').value;

        document.getElementById('user-tag').innerText = username;
        document.body.setAttribute('data-theme', theme);
        document.getElementById('persona-label').innerText = `ACTIVE: ${persona.toUpperCase()}`;
        toggleSettings();
    }

    // Initialize
    renderHistory();
    document.getElementById('set-name').value = username;
    document.getElementById('user-input').addEventListener('keypress', e => { if(e.key === 'Enter') handleSend(); });
</script>
